## Overview

MAPPA OS uses **TCP socket-based communication** for inter-process communication between modules (Kernel, CPU, Memory, Filesystem). The protocol supports structured message passing with operation codes (opcodes) and serialized data packets.

## Communication Architecture

### Module Connections

### Socket Types

**Server sockets** (listening):
```c:utils/src/sockets/sockets.c
int iniciar_servidor(t_log* logger, char* ip, char* puerto, char* nombre) {
    int socket_servidor;
    struct addrinfo hints, *servinfo;
    
    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_INET;
    hints.ai_socktype = SOCK_STREAM;
    hints.ai_flags = AI_PASSIVE;
    
    getaddrinfo(ip, puerto, &hints, &servinfo);
    
    socket_servidor = socket(servinfo->ai_family, servinfo->ai_socktype, servinfo->ai_protocol);
    int yes = 1;
    setsockopt(socket_servidor, SOL_SOCKET, SO_REUSEADDR, &yes, sizeof yes);
    
    bind(socket_servidor, servinfo->ai_addr, servinfo->ai_addrlen);
    listen(socket_servidor, SOMAXCONN);
    
    log_info(logger, "Escuchando en %s", nombre);
    freeaddrinfo(servinfo);
    
    return socket_servidor;
}
```

**Client sockets** (connecting):
```c:utils/src/sockets/sockets.c
int crear_conexion(t_log *logger, const char *server_name, char *ip, char *puerto) {
    struct addrinfo hints;
    struct addrinfo *server_info;
    
    memset(&hints, 0, sizeof(hints));
    hints.ai_family = AF_UNSPEC;
    hints.ai_socktype = SOCK_STREAM;
    hints.ai_flags = AI_PASSIVE;
    
    getaddrinfo(ip, puerto, &hints, &server_info);
    
    int socket_cliente = socket(server_info->ai_family,
                                server_info->ai_socktype,
                                server_info->ai_protocol);
    
    if (socket_cliente == -1) {
        log_error(logger, "Error creando el socket para %s:%s", ip, puerto);
        return 0;
    }
    
    if (connect(socket_cliente, server_info->ai_addr, server_info->ai_addrlen) == -1) {
        perror("Error al conectarse");
        log_error(logger, "Error al conectarme (a %s)\n", server_name);
        freeaddrinfo(server_info);
        return 0;
    } else {
        log_info(logger, "Me conecte en %s:%s (a %s)\n", ip, puerto, server_name);
    }
    
    freeaddrinfo(server_info);
    return socket_cliente;
}
```

## Message Passing

### Simple Messages

**Send:**
```c:utils/src/protocolo/protocolo.c
void enviar_mensaje(char* mensaje, int socket_cliente) {
    t_paquete* paquete = malloc(sizeof(t_paquete));
    paquete->codigo_operacion = MENSAJE;
    paquete->buffer = malloc(sizeof(t_buffer));
    paquete->buffer->size = strlen(mensaje) + 1;
    paquete->buffer->stream = malloc(paquete->buffer->size);
    memcpy(paquete->buffer->stream, mensaje, paquete->buffer->size);
    
    int bytes = paquete->buffer->size + 2*sizeof(int);
    void* a_enviar = serializar_paquete(paquete, bytes);
    
    send(socket_cliente, a_enviar, bytes, 0);
    
    free(a_enviar);
    eliminar_paquete(paquete);
}
```

**Receive:**

```c:utils/src/protocolo/protocolo.c
void recibir_mensaje(t_log* logger, int socket_cliente) {

    }
}
```

### Packet Creation

```c:utils/src/protocolo/protocolo.c
t_paquete* crear_paquete(op_code codigo_op) {
    t_paquete* paquete = malloc(sizeof(t_paquete));
    paquete->codigo_operacion = codigo_op;
    crear_buffer(paquete);
    return paquete;
}

void crear_buffer(t_paquete* paquete) {
    paquete->buffer = malloc(sizeof(t_buffer));
    paquete->buffer->size = 0;
    paquete->buffer->stream = NULL;
}
```

### Adding Data to Packets

```c:utils/src/protocolo/protocolo.c
void agregar_a_paquete(t_paquete* paquete, void* valor, int tamanio) {
    // Reallocate stream to accommodate new data
    paquete->buffer->stream = realloc(paquete->buffer->stream, 
                                     paquete->buffer->size + tamanio + sizeof(int));
    
    // Add size prefix
    memcpy(paquete->buffer->stream + paquete->buffer->size, &tamanio, sizeof(int));
    
    // Add data
    memcpy(paquete->buffer->stream + paquete->buffer->size + sizeof(int), valor, tamanio);
    
    paquete->buffer->size += tamanio + sizeof(int);
}
```

### Packet Serialization

```c:utils/src/protocolo/protocolo.c
void* serializar_paquete(t_paquete* paquete, int bytes) {
    int size_paquete = sizeof(int) + sizeof(int) + paquete->buffer->size;
    void* paquete_serializado = malloc(size_paquete);
    int offset = 0;
    
    // Serialize opcode
    memcpy(paquete_serializado + offset, &(paquete->codigo_operacion), sizeof(int));
    offset += sizeof(int);
    
    // Serialize buffer size
    memcpy(paquete_serializado + offset, &(paquete->buffer->size), sizeof(int));
    offset += sizeof(int);
    
    // Serialize buffer stream
    memcpy(paquete_serializado + offset, paquete->buffer->stream, paquete->buffer->size);
    
    return paquete_serializado; 
}
```

**Receive buffer:**
```c:utils/src/protocolo/protocolo.c
void* recibir_buffer(int* size, int socket_cliente) {
    void* buffer;
    
    recv(socket_cliente, size, sizeof(int), MSG_WAITALL);
    buffer = malloc(*size);
    recv(socket_cliente, buffer, *size, MSG_WAITALL);
    
    return buffer;
}
```

**Receive and parse packet:**
```c:utils/src/protocolo/protocolo.c
t_list* recibir_paquete(int socket_cliente) {
    int size;
    int desplazamiento = 0;
    void* buffer;
    t_list* valores = list_create();
    int tamanio;
    
    buffer = recibir_buffer(&size, socket_cliente);
    
    if (buffer == NULL) {
        list_destroy(valores);
        return NULL;
    }
    
    // Parse buffer into list of values
    while (desplazamiento < size) {
        memcpy(&tamanio, buffer + desplazamiento, sizeof(int));
        desplazamiento += sizeof(int);
        
        char* valor = malloc(tamanio);
        memcpy(valor, buffer + desplazamiento, tamanio);
        desplazamiento += tamanio;
        
        list_add(valores, valor);
    }
    
    free(buffer);
    return valores;
}
```

```c:utils/src/protocolo/protocolo.c
pcb* desempaquetar_pcb(t_list* paquete, int* counter) {
    pcb* contexto = malloc(sizeof(pcb));
    
    int* pid = list_get(paquete, 0);
    contexto->pid = *pid;
    free(pid);
    
    int* pc = list_get(paquete, 1);
    contexto->pc = *pc;
    free(pc);
    
    int* size = list_get(paquete, 2);
    contexto->size = *size;
    free(size);
    
    int* prioridad = list_get(paquete, 3);
    contexto->prioridad = *prioridad;
    free(prioridad);
    
    int* tiempo_ejecucion = list_get(paquete, 4);
    contexto->tiempo_ejecucion = *tiempo_ejecucion;
    free(tiempo_ejecucion);
    
    char* path = list_get(paquete, 5);
    contexto->path = malloc(strlen(path) + 1);
    strcpy(contexto->path, path);
    free(path);
    
    estado_proceso* estado = list_get(paquete, 6);
    contexto->estado = *estado;
    free(estado);
    
    int comienzo_registros = 7;
    t_registros* registro_contexto = desempaquetar_registros(paquete, comienzo_registros);
    contexto->registros = registro_contexto;
    
    *counter = comienzo_registros + 4;
    t_list* archivos = desempaquetar_archivos(paquete, counter);
    contexto->archivos = archivos;
    
    if (contexto->path == NULL || contexto->registros == NULL || contexto->archivos == NULL) {
        free(contexto->path);
        registros_destroy(contexto->registros);
        lista_archivos_destroy(contexto->archivos);
        free(contexto);
        return NULL;
    }
    
    return contexto;
}
```

## Communication Patterns

### Request-Response

**Example: Memory access**

```c
// CPU requests frame number
send_pedido_marco(fd_memoria, pid, pagina);
op_code response = recibir_operacion(fd_memoria);

if (response == ENVIO_MARCO) {
    int marco = recv_marco(fd_memoria);
    // Use frame number
} else if (response == PCB_PAGEFAULT) {
    // Handle page fault
}
```

### Asynchronous Notifications

**Example: Quantum interrupt**

```c:kernel/src/main.c
// Kernel sends interrupt to CPU
send_interrupcion(proceso->pid, fd_cpu_interrupt);

// CPU receives and handles interrupt
int pid_interrupcion = recv_interrupcion(fd_cpu_interrupt);
// Preempt process with matching PID
```

### One-way Messages

**Example: Process termination**

```c:kernel/src/main.c
// Kernel notifies memory to free process resources
send_terminar_proceso(proceso->pid, fd_memoria);

// Memory receives and frees resources
int pid_fin = recv_terminar_proceso(cliente_socket);
terminar_proceso(pid_fin);
```

## Thread-Safe Communication

Each module uses dedicated threads for receiving messages:

```c:memoria/src/main.c
static void procesar_conexion(void *void_args) {
    int *args = (int*) void_args;
    int cliente_socket = *args;
    op_code cop;
    
    while (cliente_socket != -1) {
        if (recv(cliente_socket, &cop, sizeof(op_code), 0) != sizeof(op_code)) {
            log_info(logger_memoria, "El cliente se desconecto");
            return;
        }
        
        switch (cop) {
            case MENSAJE:
                // Handle message
                break;
            case PEDIDO_MARCO:
                // Handle frame request
                break;
            // ... other cases
        }
    }
}
```
